				#include iostream

				#include math.h

				using namespace std;

				const double u0 = 1.0;

				double qk(double x, double k)
				{
					double result = 0.0;
					result = ((-1.0) * x * x * x) / ((2 * k + 1) * (2 * k + 2));
					return result;
				}

				double rowSum(double uk, double eps, double x, double k, double sum)
				{
					while (fabs(uk) > eps)
					{
						uk = uk * qk(x, k);
						sum = sum + uk;
						k = k + 1.0;
					}
					return sum;
				}


				int main()
				{

					vector sumVector;
					double x = 0.1;                        // точка, в которой будет высчитываться значение ряда
					double eps = 0.0001;                   // погрешность, с которой будет вычисляться значение ряда
					double uk, sum;                        // текущий член ряда и текущая сумма ряда
					double k;                              // текущий номер

					

					for (int xx = 1; xx < 17; xx = xx + 1)
					{

						uk = u0;
						sum = 0;
						k = 0.0;

						while (fabs(uk) > eps)
						{
							
							sum = sum + uk;
							uk = uk * qk(x, k);
							k = k + 1.0;
						}                                       // значение ряда будем вычислять в цикле while
						cout << sum << " ";                     // выводим сумму
