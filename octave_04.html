Описание: GNU Octave — свободная программная система для математических вычислений, использующая совместимый с MATLAB язык высокого уровня.

Интегральное исчисление возникло из задачи вычисления площадей или объемов, ограниченных кривыми линиями или поверхностями. 
			  
				a = 0;
				b = sqrt(3);
				n = 30;
				h = (b - a)/n;
				x = [a : h : b];
				y = (x.^3).*atan(x);
				f = @(x) (x.^3).*atan(x);
				epsln = 0.003;

				############################################
				It=trapz(x,y);
				fprintf('Integral trapz = %8.5f \n', It);
				############################################
				Iq = quad(f, x(1), x(length(x)));
				fprintf('Integral quad = %8.5f \n', Iq);
				############################################
				X = f(a)+f(b);
				Odd = 0;
				Even = 0;
				for i = 1:2:n-1
					xi=a+(i*h);
					Odd=Odd+f(xi);
				end
				for i = 2:2:n-2
					xi=a+(i*h);
					Even=Even+f(xi);
				end
				Is = (h/3)*(X+4*Odd+2*Even);
				fprintf('Integral Sim. = %8.5f \n', Is);
				fprintf('Function calculation accuracy = %8.5f \n', abs(Is - It));
				fprintf('Specified calculation accuracy = %8.5f \n', epsln);
				############################################
