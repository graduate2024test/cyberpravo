Описание: GNU Octave — свободная программная система для математических вычислений, использующая совместимый с MATLAB язык высокого уровня.

Метод бисекции или метод деления отрезка пополам — простейший численный метод для решения нелинейных уравнений вида f(x)=0. Предполагается только непрерывность функции f(x). Поиск основывается на теореме о промежуточных значениях. 
			  
				function bisection_method()
					f = @(x) x^3 - e^(4*x) + 5.5;
					eps = 1e-3;
					a = -1;   b = 3;
					[x0, fx0] = bisection(@fun01, a, b, eps);
					if x0 <= b
						fprintf('F(x): %d\n', fx0);
						fprintf('x   : %d\n', x0);
					else
						fprintf('Abort execution.\n');
					endif
				endfunction

				function [result1, result2] = bisection(f, x_L, x_R, eps)
					if f(x_L)*f(x_R) > 0
						fprintf('The function has no solutions on the interval!\n')
					endif
					x_M = (x_L + x_R)/2.0;
					f_M = f(x_M);
					iteration_counter = 1;
					while abs(f_M) > eps
						left_f = f(x_L);
						right_f = f(x_R);
						if left_f*f_M > 0
							x_L = x_M;
						else
							x_R = x_M;
						endif
						x_M = (x_L + x_R)/2;
						f_M = f(x_M);
						if iteration_counter > 100
						  break;
						endif
						iteration_counter = iteration_counter + 1;
					endwhile
					result1 = x_M;
					result2 = f_M;
				endfunction
