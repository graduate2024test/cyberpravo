Формула прямоугольников.
Метод трапеций.

				#include iostream
				#include math.h
				using namespace std;

				const double a = exp(1);
				const double b = exp(2);


				//подынтегральная функция
				double InFunction(double x) {                            
					return (x*log(x));
				}


				//формула средних прямоугольников для аналитически заданной функции левых
				double CalcIntegral1(double a, double b, int n)
				{                                                       
					double result = 0, h = (b - a) / n;

					for (int i = 0; i < n; i++) {
						result += InFunction(a + h * (i + 0.5));
					}

					result *= h;
					return result;
				}        


				//формула средних прямоугольников для аналитически заданной функции правых
				double CalcIntegral2(double a, double b, int n)
				{                                                      
					double result = 0, h = (b - a) / n;

					for (int i = 0; i < n; i++) {
						result += InFunction(a + h * i);
					}

					result *= h;
					return result;
				}


				//формула средних прямоугольников для аналитически заданной функции средних
				double CalcIntegral3(double a, double b, int n)
				{                                                       
					double result = 0, h = (b - a) / n;

					for (int i = 0; i < n; i++) {
						result += InFunction(a + h * (i + 1));
					}

					result *= h;
					return result;
				}


				//метод трапеций
				double TrapezoidalIntegral(double a, double b, int n) 
				{                                                       
					const double width = (b - a) / n;

					double trapezoidal_integral = 0;
					for (int step = 0; step < n; step++) {
						const double x1 = a + step * width;
						const double x2 = a + (step + 1) * width;

						trapezoidal_integral += 0.5 * (x2 - x1) *
							(InFunction(x1) + InFunction(x2));
					}

					return trapezoidal_integral;
				}

				
				//метод Симпсона
				double SimpsonIntegral(double a, double b, int n)
				{                                                       
					const double width = (b - a) / n;

					double simpson_integral = 0;
					for (int step = 0; step < n; step++) {
						const double x1 = a + step * width;
						const double x2 = a + (step + 1) * width;

						simpson_integral += (x2 - x1) / 6.0 * (InFunction(x1) + 4.0 *
							InFunction(0.5 * (x1 + x2)) + InFunction(x2));
					}

					return simpson_integral;
				}

				int main()
				{
					double result11 = CalcIntegral1(a, b, 50);
					cout << "Medium Rectangle Formula = " << result11;
					cout << endl;

					double result12 = CalcIntegral2(a, b, 50);
					cout << "Left Rectangle Formula = " << result12;
					cout << endl;

					double result13 = CalcIntegral3(a, b, 50);
					cout << "Right Rectangle Formula = " << result13;
					cout << endl;

					double result2 = TrapezoidalIntegral(a, b, 50);
					cout << "Trapezoidal Integral = " << result2;
					cout << endl;

					double result3 = SimpsonIntegral(a, b, 50);
					cout << "Simpson Integral = " << result3;
					cout << endl;


				}
