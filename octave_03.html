				x = [-6.0 -4.0 -2.0 0.0 2.0 4.0 6.0]
				y = [-5.76 -2.24 -0.48 0.0 -0.32 -0.96 -1.44]
				y_true = 0.01* (x ).^ 3- 0.4* (x / 2).^ 2

				fprintf("Linear interpolation using Octave\n")
				subplot(2, 2, 1)
				p2 = polyfit(x,y,2);
				f2 = polyval(p2,x);
				plot(x,f2,'-m')
				title("Linear Octave\n")
				grid on

				fprintf("Cubic interpolation using Octave\n")
				subplot(2, 2, 2)
				p3 = polyfit(x,y,3);
				f3 = polyval(p3,x);
				plot(x,f3,':b')
				title("Cubic Octave\n")
				grid on

				fprintf("Spline interpolation using Octave\n")
				subplot(2, 2, 3)
				xsi = x(1):0.25:x(length(x))
				ps = spline(x,y);
				fsi = ppval(ps,xsi);
				plot(xsi,fsi,':g')
				title("Spline Octave\n")
				grid on

				fprintf("True function\n")
				subplot(2, 2, 4)
				plot(x,y_true,'or')
				title("True function\n")
				legend('0.01* x^3- 0.4* (x / 2)^2', "location", "south");
				grid on


				hold off
