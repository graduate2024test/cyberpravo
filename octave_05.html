Дифференциальное исчисление — это неотъемлемая часть анализа и, следовательно, раздел математики . Его центральная тема — вычисление локальных изменений функций . В то время как непрерывная функция постоянно присваивает определенные выходные значения своим входным значениям, дифференциальное исчисление определяет, насколько изменяются выходные значения после очень малых изменений входных значений. 
			  
				##pkg load symbolic;

				syms x;

				h = 0.1;
				x = (0:h:3);
				y = sin(x).^2;
				dy = 2*sin(x).*cos(x);
				#########################################
				n = length(x);
				% rigth ---------------------------
				for i = 1:n-1
				  dy_r(i) = (sin(x(i + 1))^2- sin(x(i))^2)/h;
				  er_r(i) = abs(dy_r(i) - dy(i));
				end
				% left ------------------------
				dy_l(1) = 0;
				er_l(1) = 0;
				for i = 2:n
				  dy_l(i) = (sin(x(i))^2- sin(x(i - 1))^2)/h;
				  er_l(i) = abs(dy_l(i) - dy(i));
				end
				% central---------------------
				dy_c(1) = 0;
				er_c(1) = 0;
				for i = 2:n - 1
				  dy_c(i) = (sin(x(i + 1))^2- sin(x(i - 1))^2)/(2*h);
				  er_c(i) = abs(dy_c(i) - dy(i));
				end
				subplot(2, 2, 2)
				plot(x,dy,'-o')
				title('Точная производная')
				subplot(2, 2, 1)
				plot(x([1:n-1]),dy_r([1:n-1]),'-h',x([2:n]),dy_l([2:n]),'-p',x([2:n-1]),dy_c([2:n-1]),'-*');
				legend('правые разности', 'левые разности', 'центральные разности ');
				title('Производные');
				#########################################
				h1 = 0.2;
				x1 = (0:h:3);
				y1 = sin(x).^2;
				dy1 = 2*sin(x).*cos(x);
				% central---------------------
				dy_c2(1) = 0;

				for i = 2:n - 1
				  dy_c2(i) = (sin(x(i + 1))^2- sin(x(i - 1))^2)/(2*h);
				  er_c2(i) = abs(dy_c(i) - dy(i));
				end
				subplot(2, 2, 3)
				plot(x([1:n-1]),dy_c2([1:n-1]), '-h');
				title('Центральные разности');
				#########################################
				h2 = 0.1;
				x2 = 0:h2:pi;
				n2 = length(x2);
				h_m = h2/2;
				x_m = 0:h_m:pi;

				  dy_C(i) = (sin(x(i+1))^2-sin(x(i))^2)/h2;
				  er1(i) = abs((2*sin(x2(i))*cos(x2(i)))-dy_C(i));
				endfor

				for i = 1:(m-1)
				  dym(i) = (sin(x_m(i+1))-sin(x_m(i)))/h_m;
				  er2(i) = abs((2*sin(x_m(i))*cos(x_m(i)))-dym(i));
				endfor

				%уточняем значение производной с помощью метода Рунге
				for i = 1:(n-1)
				  dy_R(i) = dy(i)-2*(dy(i)-dym(2*i-1));
				  er3(i) = abs((2*sin(x2(i))*cos(x2(i)))-dy_R(i));
				endfor
